#include <iostream> 
#include <vector> 
#include <stack> 
#include <cctype> 
using namespace std; 
 
// Check if character is an operator 
bool isOperator(char c) 
{ 
    return c == '+' || c == '-' || c == '*' || 
           c == '/' || c == '%' || c == '!'; 
} 
 
// ----------- Tokenization ----------- 
vector<string> tokenize(const string &line) 
{ 
    vector<string> tokens; 
    string token = ""; 
 
    for (char c : line + " ") 
    { // Extra space to flush last token 
        if (isalnum(c) || c == '_') 
        { 
            token += c; // Build operand 
        } 
        else 
        { 
            if (!token.empty()) 
            { 
                tokens.push_back(token); 
                token.clear(); 
            } 
 
            if (isspace(c)) 
                continue; 
            else if (isOperator(c) || c == '(' || c == ')') 
                tokens.push_back(string(1, c)); 
            else 
                throw runtime_error("Invalid character"); 
        } 
    } 
    return tokens; 
} 
 
// ----------- Syntax & Parentheses Validation ----------- 
bool isValidExpression(const vector<string> &tokens) 
{ 
    stack<char> st; 
 
    for (int i = 0; i < tokens.size(); i++) 
    { 
        string t = tokens[i]; 
 
        // Parentheses handling 
        if (t == "(") 
            st.push('('); 
        else if (t == ")") 
        { 
            if (st.empty()) 
                return false; 
            st.pop(); 
        } 
 
        // Operator validation 
        if (t.size() == 1 && isOperator(t[0])) 
        { 
            char op = t[0]; 
 
            // Cannot be the last token 
            if (i == tokens.size() - 1) 
                return false; 
 
            // Determine if unary 
            bool unary = (op == '+' || op == '-' || op == '!') && 
                         (i == 0 || 
                          tokens[i - 1] == "(" || 
                          (tokens[i - 1].size() == 1 && 
isOperator(tokens[i - 1][0]))); 
 
            if (!unary) 
            { 
                // Binary operator must be between valid operands or 
')' 
                if (i == 0 || 
                    !(isalnum(tokens[i - 1][0]) || tokens[i - 1] == 
")") || 
                    !(isalnum(tokens[i + 1][0]) || tokens[i + 1] == 
"(")) 
                    return false; 
            } 
        } 
 
        // Two operands together → invalid 
        if (i > 0 && 
            isalnum(tokens[i][0]) && 
            isalnum(tokens[i - 1][0])) 
            return false; 
 
        // Operand followed by '(' → invalid 
        if (i > 0 && 
            tokens[i] == "(" && 
            isalnum(tokens[i - 1][0])) 
            return false; 
 
        // ')' followed by operand → invalid 
        if (i > 0 && 
            isalnum(tokens[i][0]) && 
            tokens[i - 1] == ")") 
            return false; 
    } 
 
    return st.empty(); // Ensure parentheses are balanced 
} 
 
int main() 
{ 
    string expr; 
    cout << "Enter expression: ";

        // a) Display tokens

        else

    } 
    catch (...) 
    { 
        cout << "Result: INVALID (Invalid character)\n"; 
    }

5. Write a program to input a context-free grammar and compute 
the FIRST and FOLLOW sets by applying the rules. Show the 
results clearly for each non-terminal. 
• Understand the concept of FIRST and FOLLOW in syntax analysis 
• Implement algorithms for computing FIRST and FOLLOW sets 
• Verify whether the grammar is suitable for LL (1) parsing 
 
#include <bits/stdc++.h> 
using namespace std; 
map<char, vector<string>> grammar; 
map<char, set<char>> FIRST, FOLLOW; 
set<char> nonTerminals; 
char startSymbol; 
/* FIRST */ 
void findFirst(char symbol) { 
for (string prod : grammar[symbol]) { 
// Terminal 
if (!isupper(prod[0]) && prod[0] != '#') { 
FIRST[symbol].insert(prod[0]); 
} 
// Epsilon 
else if (prod[0] == '#') { 
FIRST[symbol].insert('#'); 
} 
// Non-terminal 
        else { 
            if (FIRST[prod[0]].empty()) 
                findFirst(prod[0]); 
 
            for (char c : FIRST[prod[0]]) { 
                if (c != '#') 
                    FIRST[symbol].insert(c); 
            } 
 
            // If epsilon is present in FIRST of next symbol 
            if (FIRST[prod[0]].count('#')) 
                FIRST[symbol].insert('#'); 
        } 
    } 
} 
 
/* FOLLOW */ 
void findFollow(char symbol) { 
 
    // Rule 1: Add $ to FOLLOW of start symbol 
    if (symbol == startSymbol) 
        FOLLOW[symbol].insert('$'); 
 
    for (auto g : grammar) { 
        char lhs = g.first; 
 
        for (string prod : g.second) { 
            for (int i = 0; i < prod.length(); i++) { 
 
                if (prod[i] == symbol) { 
 
                    // Next symbol exists 
                    if (i + 1 < prod.length()) { 
                        char next = prod[i + 1]; 
 
                        if (!isupper(next) && next != '#') { 
                            FOLLOW[symbol].insert(next); 
                        } else { 
                            for (char c : FIRST[next]) { 
                                if (c != '#') 
                                    FOLLOW[symbol].insert(c); 
                            } 
                            // If epsilon in FIRST(next), add 
FOLLOW(lhs) 
                            if (FIRST[next].count('#')) { 
                                for (char c : FOLLOW[lhs]) 
                                    FOLLOW[symbol].insert(c); 
                            } 
                        } 
                    } 
                    // End of production 
                    else if (lhs != symbol) { 
                        for (char c : FOLLOW[lhs]) 
                            FOLLOW[symbol].insert(c); 
                    } 
                } 
            } 
        } 
    } 
} 
 
/* LL(1) Check */ 
bool checkLL1() { 
    for (char nt : nonTerminals) { 
 
        vector<set<char>> firstSets; 
        bool hasEpsilon = false; 
 
        for (string prod : grammar[nt]) { 
            set<char> temp; 
 
            if (prod[0] == '#') { 
                hasEpsilon = true; 
            } 
            else if (!isupper(prod[0])) { 
                temp.insert(prod[0]); 
            } 
            else { 
                for (char c : FIRST[prod[0]]) { 
                    if (c != '#') temp.insert(c); 
                } 
                if (FIRST[prod[0]].count('#')) 
                    hasEpsilon = true; 
            } 
 
            firstSets.push_back(temp); 
        } 
 
        // FIRST/FIRST conflict 
        for (int i = 0; i < firstSets.size(); i++) { 
            for (int j = i + 1; j < firstSets.size(); j++) { 
                for (char c : firstSets[i]) { 
                    if (firstSets[j].count(c)) 
                        return false; 
                } 
            } 
        } 
 
        // FIRST/FOLLOW conflict when epsilon exists 
        if (hasEpsilon) { 
            for (auto fs : firstSets) { 
                for (char c : fs) { 
                    if (FOLLOW[nt].count(c)) 
                        return false; 
                } 
            } 
        } 
    } 
    return true; 
} 
 
int main() { 
    int n; 
    cout << "Enter number of productions: "; 
    cin >> n; 
 
    cout << "Enter productions (Example: E=TR|a):\n"; 
    for (int i = 0; i < n; i++) { 
        string prod; 
        cin >> prod; 
 
        char lhs = prod[0]; 
        string rhs = prod.substr(2); 
        nonTerminals.insert(lhs); 
 
        if (i == 0) 
            startSymbol = lhs; 
 
        // Split RHS using '|' 
        string temp = ""; 
        for (char c : rhs) { 
            if (c == '|') { 
                grammar[lhs].push_back(temp); 
                temp = ""; 
            } else { 
                temp += c; 
            } 
        } 
        grammar[lhs].push_back(temp); 
    } 
 
    // Compute FIRST 
    for (char nt : nonTerminals) 
        findFirst(nt); 
 
    // Compute FOLLOW (iterate to propagate) 
    for (int i = 0; i < nonTerminals.size(); i++) { 
        for (char nt : nonTerminals) 
            findFollow(nt); 
    } 
 
    // Display FIRST sets 
    cout << "\nFIRST Sets:\n"; 
    for (char nt : nonTerminals) { 
        cout << "FIRST(" << nt << ") = { "; 
        for (char c : FIRST[nt]) cout << c << " "; 
        cout << "}\n"; 
    } 
 
    // Display FOLLOW sets 
    cout << "\nFOLLOW Sets:\n"; 
    for (char nt : nonTerminals) { 
        cout << "FOLLOW(" << nt << ") = { "; 
        for (char c : FOLLOW[nt]) cout << c << " "; 
        cout << "}\n"; 
    } 
 
    // LL(1) Check 
    cout << "\nLL(1) Check:\n"; 
    if (checkLL1()) 
        cout << "Grammar IS LL(1)\n"; 
    else 
        cout << "Grammar is NOT LL(1)\n"; 
 
    return 0; 
}

#include <iostream> 
#include <map> 
#include <vector> 
#include <set> 
#include <cctype> 
using namespace std; 
map<char, vector<string>> grammar, cnf; 
map<char, char> termMap; // Maps terminals to new variables 
char startSymbol; 
int varCount = 0; 
// Step 1: Replace terminals in long productions 
void replaceTerminals() 
{ 
for (map<char, vector<string>>::iterator it = grammar.begin(); 
it != grammar.end(); ++it) 
{ 
char lhs = it->first; 
vector<string> rules = it->second; 
        for (int i = 0; i < rules.size(); i++) 
        { 
            string rule = rules[i]; 
 
            if (rule.length() > 1) 
            { 
                for (int j = 0; j < rule.length(); j++) 
                { 
                    if (islower(rule[j])) 
                    { 
                        if (!termMap.count(rule[j])) 
                        { 
                            char newVar = 'T' + varCount++; // T, U, 
V... 
                            termMap[rule[j]] = newVar; 
                            cnf[newVar].push_back(string(1, rule[j])); 
                        } 
                        rule[j] = termMap[rule[j]]; 
                    } 
                } 
            } 
            cnf[lhs].push_back(rule); 
        } 
    } 
} 
 
// Step 2: Break long productions into binary rules 
void breakLongRules() 
{ 
    map<char, vector<string>> newCNF; 
    int count = 0; 
 
    for (map<char, vector<string>>::iterator it = cnf.begin(); 
         it != cnf.end(); ++it) 
    { 
 
        char lhs = it->first; 
        vector<string> rules = it->second; 
 
        for (int i = 0; i < rules.size(); i++) 
        { 
            string rule = rules[i]; 
 
            if (rule.length() <= 2) 
            { 
                newCNF[lhs].push_back(rule); 
            } 
            else 
            { 
                char prev = lhs; 
 
                for (int j = 0; j < rule.length() - 2; j++) 
                { 
                    char newVar = 'X' + count++; 
                    newCNF[prev].push_back(string() + rule[j] + 
newVar); 
                    prev = newVar; 
                } 
                newCNF[prev].push_back( 
                    string() + rule[rule.length() - 2] + rule.back()); 
            } 
        } 
    } 
    cnf = newCNF; 
} 
 
// CYK Algorithm 
bool CYK(string input) 
{ 
    int n = input.length(); 
    vector<vector<set<char>>> dp(n, vector<set<char>>(n)); 
 
    // Step 1: Fill diagonal for A -> a 
    for (int i = 0; i < n; i++) 
    { 
        for (map<char, vector<string>>::iterator it = cnf.begin(); 
             it != cnf.end(); ++it) 
        { 
 
            char lhs = it->first; 
            vector<string> rules = it->second; 
 
            for (int j = 0; j < rules.size(); j++) 
            { 
                if (rules[j].length() == 1 && 
                    rules[j][0] == input[i]) 
                { 
                    dp[i][i].insert(lhs); 
                } 
            } 
        } 
    } 
 
    // Step 2: Fill table for substrings of length >= 2 
    for (int len = 2; len <= n; len++) 
    { 
        for (int i = 0; i <= n - len; i++) 
        { 
            int j = i + len - 1; 
 
            for (int k = i; k < j; k++) 
            { 
                for (map<char, vector<string>>::iterator it = 
cnf.begin(); 
                     it != cnf.end(); ++it) 
                { 
 
                    char lhs = it->first; 
                    vector<string> rules = it->second; 
 
                    for (int r = 0; r < rules.size(); r++) 
                    { 
                        if (rules[r].length() == 2) 
                        { 
                            char B = rules[r][0]; 
                            char C = rules[r][1]; 
 
                            if (dp[i][k].count(B) && 
                                dp[k + 1][j].count(C)) 
                            { 
                                dp[i][j].insert(lhs); 
                            } 
                        } 
                    } 
                } 
            } 
        } 
    } 
 
    // Accept if start symbol derives the whole string 
    return dp[0][n - 1].count(startSymbol); 
}

8. Infix to Postfix Translation using SDT 
Scenario: 
A compiler for a stack-based virtual machine requires expressions to 
be converted into postfix notation before execution. The developer 
uses Syntax Directed Translation to convert infix expressions into 
postfix form during parsing. 
Task: 
Design an SDT scheme to translate infix expressions into postfix 
expressions. 
Requirements: 
1. Write CFG for arithmetic expressions. 
2. Attach semantic rules to generate postfix output. 
3. Test using inputs: 
4. A + B * C 
5. (A + B) * C 
6. A * B + C / D 
7. Display postfix expression. 
#include <iostream> 
#include <stack> 
#include <cctype> 
using namespace std; 
 
int prec(char op) 
{ 
    if (op == '+' || op == '-') 
        return 1; 
    if (op == '*' || op == '/') 
        return 2; 
    return 0; 
} 
 
int main() 
{ 
    string infix, postfix = ""; 
    stack<char> st; 
 
    cout << "Enter infix expression: "; 
    getline(cin, infix); 
 
    for (char c : infix) 
    { 
        if (isspace(c)) 
            continue; 
 
        // Operand 
        if (isalnum(c)) 
        { 
            postfix += c; 
            postfix += ' '; 
        } 
        // Opening parenthesis 
        else if (c == '(') 
        { 
            st.push(c); 
        } 
        // Closing parenthesis 
        else if (c == ')') 
        { 
            while (!st.empty() && st.top() != '(') 
            { 
                postfix += st.top(); 
                postfix += ' '; 
                st.pop(); 
            } 
            if (!st.empty()) 
                st.pop(); // Remove '(' 
        } 
        // Operator 
        else 
        { 
            while (!st.empty() && prec(st.top()) >= prec(c)) 
            { 
                postfix += st.top(); 
                postfix += ' '; 
                st.pop(); 
            } 
            st.push(c); 
        } 
    } 
 
    // Pop remaining operators 
    while (!st.empty()) 
    { 
        postfix += st.top(); 
        postfix += ' '; 
        st.pop(); 
    } 
 
    cout << "Postfix Expression: " << postfix << endl; 
    return 0; 
}

// Convert Infix to Postfix 
string infixToPostfix(const string &infix) 
{ 
    stack<char> st; 
    string postfix = ""; 
 
    for (char c : infix) 
    { 
        if (isspace(c)) 
            continue; // Ignore spaces 
 
        if (isalnum(c)) 
        { // Operand 
            postfix += c; 
        } 
        else if (c == '(') 
        { // Opening parenthesis 
            st.push(c); 
        } 
        else if (c == ')') 
        { // Closing parenthesis 
            while (!st.empty() && st.top() != '(') 
            { 
                postfix += st.top(); 
                st.pop(); 
            } 
            if (!st.empty()) 
                st.pop(); // Remove '(' 
        } 
        else 
        { // Operator 
            while (!st.empty() && precedence(st.top()) >= 
precedence(c)) 
            { 
                postfix += st.top(); 
                st.pop(); 
            } 
            st.push(c); 
        } 
    } 
 
    // Pop remaining operators 
    while (!st.empty()) 
    { 
        postfix += st.top(); 
        st.pop(); 
    } 
 
    return postfix; 
} 
 
// Generate Three Address Code from Postfix 
void generateTAC(const string &postfix, char lhs) 
{ 
    stack<string> st; 
    int tempCount = 1; 
    vector<string> tac; // Store TAC instructions 
 
    for (char c : postfix) 
    { 
        if (isalnum(c)) 
        { 
            st.push(string(1, c)); // Push operand 
        } 
        else 
        { 
            string op2 = st.top(); 
            st.pop(); 
            string op1 = st.top(); 
            st.pop(); 
 
            string temp = "t" + to_string(tempCount++); 
            tac.push_back(temp + " = " + op1 + " " + c + " " + op2); 
            st.push(temp); 
        } 
    } 
 
    // Final assignment 
    tac.push_back(string(1, lhs) + " = " + st.top()); 
 
    // Display TAC 
    cout << "\nThree Address Code:\n"; 
    for (const string &line : tac) 
        cout << line << endl; 
} 
 
int main() 
{ 
    string input, expr; 
    char lhs; 
 
    cout << "Enter expression (e.g., c=(a+b)*(c+d)):\n"; 
    cin >> input; 
 
lhs = input[0];                           
// Left-hand side 
variable 
expr = input.substr(input.find('=') + 1); // Extract RHS expression 
string postfix = infixToPostfix(expr); 
generateTAC(postfix, lhs); 
return 0; 
}

void backpatch(const vector<int> &list, int target) 
{

        cout << "Instruction " << list[i] 
             << " patched with target " << target << endl; 
}
